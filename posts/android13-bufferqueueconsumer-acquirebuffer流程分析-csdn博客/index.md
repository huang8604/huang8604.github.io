# Android13 BufferQueueConsumer AcquireBuffer流程分析-CSDN博客

BufferQueueConsumer的acquireBuffer方法的主要作用是从BufferQueue中获取一个可用的图像缓冲区，并返回一个GraphicBuffer对象。它可以用于在应用程序中进行图像处理、渲染或显示等操作。

在调用acquireBuffer方法时，它会首先检查是否有可用的图像缓冲区。如果有可用的缓冲区，则会将其标记为“已使用”，并返回该缓冲区的GraphicBuffer对象。如果没有可用的缓冲区，则会等待直到有可用的缓冲区为止。

BufferQueueConsumer的acquireBuffer方法代码如下：

```cpp
//frameworks/native/libs/gui/BufferQueueConsumer.cpp
status_t BufferQueueConsumer::acquireBuffer(BufferItem* outBuffer,
        nsecs_t expectedPresent, uint64_t maxFrameNumber) {
    ATRACE_CALL();
 
 
    int numDroppedBuffers = 0;
    sp&lt;IProducerListener&gt; listener;
    {
        std::unique_lock&lt;std::mutex&gt; lock(mCore-&gt;mMutex);
 
 
        // Check that the consumer doesn&#39;t currently have the maximum number of
        // buffers acquired. We allow the max buffer count to be exceeded by one
        // buffer so that the consumer can successfully set up the newly acquired
        // buffer before releasing the old one.
        int numAcquiredBuffers = 0;
        for (int s : mCore-&gt;mActiveBuffers) {
            if (mSlots[s].mBufferState.isAcquired()) {
                &#43;&#43;numAcquiredBuffers;
            }
        }
        const bool acquireNonDroppableBuffer = mCore-&gt;mAllowExtraAcquire &amp;&amp;
                numAcquiredBuffers == mCore-&gt;mMaxAcquiredBufferCount &#43; 1;
        if (numAcquiredBuffers &gt;= mCore-&gt;mMaxAcquiredBufferCount &#43; 1 &amp;&amp;
            !acquireNonDroppableBuffer) {
            BQ_LOGE(&#34;acquireBuffer: max acquired buffer count reached: %d (max %d)&#34;,
                    numAcquiredBuffers, mCore-&gt;mMaxAcquiredBufferCount);
            return INVALID_OPERATION;
        }
 
 
        bool sharedBufferAvailable = mCore-&gt;mSharedBufferMode &amp;&amp;
                mCore-&gt;mAutoRefresh &amp;&amp; mCore-&gt;mSharedBufferSlot !=
                BufferQueueCore::INVALID_BUFFER_SLOT;
 
 
        // In asynchronous mode the list is guaranteed to be one buffer deep,
        // while in synchronous mode we use the oldest buffer.
        if (mCore-&gt;mQueue.empty() &amp;&amp; !sharedBufferAvailable) {
            return NO_BUFFER_AVAILABLE;
        }
 
 
        BufferQueueCore::Fifo::iterator front(mCore-&gt;mQueue.begin());
 
 
        // If expectedPresent is specified, we may not want to return a buffer yet.
        // If it&#39;s specified and there&#39;s more than one buffer queued, we may want
        // to drop a buffer.
        // Skip this if we&#39;re in shared buffer mode and the queue is empty,
        // since in that case we&#39;ll just return the shared buffer.
        if (expectedPresent != 0 &amp;&amp; !mCore-&gt;mQueue.empty()) {
            // The &#39;expectedPresent&#39; argument indicates when the buffer is expected
            // to be presented on-screen. If the buffer&#39;s desired present time is
            // earlier (less) than expectedPresent -- meaning it will be displayed
            // on time or possibly late if we show it as soon as possible -- we
            // acquire and return it. If we don&#39;t want to display it until after the
            // expectedPresent time, we return PRESENT_LATER without acquiring it.
            //
            // To be safe, we don&#39;t defer acquisition if expectedPresent is more
            // than one second in the future beyond the desired present time
            // (i.e., we&#39;d be holding the buffer for a long time).
            //
            // NOTE: Code assumes monotonic time values from the system clock
            // are positive.
 
 
            // Start by checking to see if we can drop frames. We skip this check if
            // the timestamps are being auto-generated by Surface. If the app isn&#39;t
            // generating timestamps explicitly, it probably doesn&#39;t want frames to
            // be discarded based on them.
            while (mCore-&gt;mQueue.size() &gt; 1 &amp;&amp; !mCore-&gt;mQueue[0].mIsAutoTimestamp) {
                const BufferItem&amp; bufferItem(mCore-&gt;mQueue[1]);
 
 
                // If dropping entry[0] would leave us with a buffer that the
                // consumer is not yet ready for, don&#39;t drop it.
                if (maxFrameNumber &amp;&amp; bufferItem.mFrameNumber &gt; maxFrameNumber) {
                    break;
                }
 
 
                // If entry[1] is timely, drop entry[0] (and repeat). We apply an
                // additional criterion here: we only drop the earlier buffer if our
                // desiredPresent falls within &#43;/- 1 second of the expected present.
                // Otherwise, bogus desiredPresent times (e.g., 0 or a small
                // relative timestamp), which normally mean &#34;ignore the timestamp
                // and acquire immediately&#34;, would cause us to drop frames.
                //
                // We may want to add an additional criterion: don&#39;t drop the
                // earlier buffer if entry[1]&#39;s fence hasn&#39;t signaled yet.
                nsecs_t desiredPresent = bufferItem.mTimestamp;
                if (desiredPresent &lt; expectedPresent - MAX_REASONABLE_NSEC ||
                        desiredPresent &gt; expectedPresent) {
                    // This buffer is set to display in the near future, or
                    // desiredPresent is garbage. Either way we don&#39;t want to drop
                    // the previous buffer just to get this on the screen sooner.
                    BQ_LOGV(&#34;acquireBuffer: nodrop desire=%&#34; PRId64 &#34; expect=%&#34;
                            PRId64 &#34; (%&#34; PRId64 &#34;) now=%&#34; PRId64,
                            desiredPresent, expectedPresent,
                            desiredPresent - expectedPresent,
                            systemTime(CLOCK_MONOTONIC));
                    break;
                }
 
 
                BQ_LOGV(&#34;acquireBuffer: drop desire=%&#34; PRId64 &#34; expect=%&#34; PRId64
                        &#34; size=%zu&#34;,
                        desiredPresent, expectedPresent, mCore-&gt;mQueue.size());
 
 
                if (!front-&gt;mIsStale) {
                    // Front buffer is still in mSlots, so mark the slot as free
                    mSlots[front-&gt;mSlot].mBufferState.freeQueued();
 
 
                    // After leaving shared buffer mode, the shared buffer will
                    // still be around. Mark it as no longer shared if this
                    // operation causes it to be free.
                    if (!mCore-&gt;mSharedBufferMode &amp;&amp;
                            mSlots[front-&gt;mSlot].mBufferState.isFree()) {
                        mSlots[front-&gt;mSlot].mBufferState.mShared = false;
                    }
 
 
                    // Don&#39;t put the shared buffer on the free list
                    if (!mSlots[front-&gt;mSlot].mBufferState.isShared()) {
                        mCore-&gt;mActiveBuffers.erase(front-&gt;mSlot);
                        mCore-&gt;mFreeBuffers.push_back(front-&gt;mSlot);
                    }
 
 
                    if (mCore-&gt;mBufferReleasedCbEnabled) {
                        listener = mCore-&gt;mConnectedProducerListener;
                    }
                    &#43;&#43;numDroppedBuffers;
                }
 
 
                mCore-&gt;mQueue.erase(front);
                front = mCore-&gt;mQueue.begin();
            }
 
 
            // See if the front buffer is ready to be acquired
            nsecs_t desiredPresent = front-&gt;mTimestamp;
            bool bufferIsDue = desiredPresent &lt;= expectedPresent ||
                    desiredPresent &gt; expectedPresent &#43; MAX_REASONABLE_NSEC;
            bool consumerIsReady = maxFrameNumber &gt; 0 ?
                    front-&gt;mFrameNumber &lt;= maxFrameNumber : true;
            if (!bufferIsDue || !consumerIsReady) {
                BQ_LOGV(&#34;acquireBuffer: defer desire=%&#34; PRId64 &#34; expect=%&#34; PRId64
                        &#34; (%&#34; PRId64 &#34;) now=%&#34; PRId64 &#34; frame=%&#34; PRIu64
                        &#34; consumer=%&#34; PRIu64,
                        desiredPresent, expectedPresent,
                        desiredPresent - expectedPresent,
                        systemTime(CLOCK_MONOTONIC),
                        front-&gt;mFrameNumber, maxFrameNumber);
                ATRACE_NAME(&#34;PRESENT_LATER&#34;);
                return PRESENT_LATER;
            }
 
 
            BQ_LOGV(&#34;acquireBuffer: accept desire=%&#34; PRId64 &#34; expect=%&#34; PRId64 &#34; &#34;
                    &#34;(%&#34; PRId64 &#34;) now=%&#34; PRId64, desiredPresent, expectedPresent,
                    desiredPresent - expectedPresent,
                    systemTime(CLOCK_MONOTONIC));
        }
 
 
        int slot = BufferQueueCore::INVALID_BUFFER_SLOT;
 
 
        if (sharedBufferAvailable &amp;&amp; mCore-&gt;mQueue.empty()) {
            // make sure the buffer has finished allocating before acquiring it
            mCore-&gt;waitWhileAllocatingLocked(lock);
 
 
            slot = mCore-&gt;mSharedBufferSlot;
 
 
            // Recreate the BufferItem for the shared buffer from the data that
            // was cached when it was last queued.
            outBuffer-&gt;mGraphicBuffer = mSlots[slot].mGraphicBuffer;
            outBuffer-&gt;mFence = Fence::NO_FENCE;
            outBuffer-&gt;mFenceTime = FenceTime::NO_FENCE;
            outBuffer-&gt;mCrop = mCore-&gt;mSharedBufferCache.crop;
            outBuffer-&gt;mTransform = mCore-&gt;mSharedBufferCache.transform &amp;
                    ~static_cast&lt;uint32_t&gt;(
                    NATIVE_WINDOW_TRANSFORM_INVERSE_DISPLAY);
            outBuffer-&gt;mScalingMode = mCore-&gt;mSharedBufferCache.scalingMode;
            outBuffer-&gt;mDataSpace = mCore-&gt;mSharedBufferCache.dataspace;
            outBuffer-&gt;mFrameNumber = mCore-&gt;mFrameCounter;
            outBuffer-&gt;mSlot = slot;
            outBuffer-&gt;mAcquireCalled = mSlots[slot].mAcquireCalled;
            outBuffer-&gt;mTransformToDisplayInverse =
                    (mCore-&gt;mSharedBufferCache.transform &amp;
                    NATIVE_WINDOW_TRANSFORM_INVERSE_DISPLAY) != 0;
            outBuffer-&gt;mSurfaceDamage = Region::INVALID_REGION;
            outBuffer-&gt;mQueuedBuffer = false;
            outBuffer-&gt;mIsStale = false;
            outBuffer-&gt;mAutoRefresh = mCore-&gt;mSharedBufferMode &amp;&amp;
                    mCore-&gt;mAutoRefresh;
        } else if (acquireNonDroppableBuffer &amp;&amp; front-&gt;mIsDroppable) {
            BQ_LOGV(&#34;acquireBuffer: front buffer is not droppable&#34;);
            return NO_BUFFER_AVAILABLE;
        } else {
            slot = front-&gt;mSlot;
            *outBuffer = *front;
        }
 
 
        ATRACE_BUFFER_INDEX(slot);
 
 
        BQ_LOGV(&#34;acquireBuffer: acquiring { slot=%d/%&#34; PRIu64 &#34; buffer=%p }&#34;,
                slot, outBuffer-&gt;mFrameNumber, outBuffer-&gt;mGraphicBuffer-&gt;handle);
 
 
        if (!outBuffer-&gt;mIsStale) {
            mSlots[slot].mAcquireCalled = true;
            // Don&#39;t decrease the queue count if the BufferItem wasn&#39;t
            // previously in the queue. This happens in shared buffer mode when
            // the queue is empty and the BufferItem is created above.
            if (mCore-&gt;mQueue.empty()) {
                mSlots[slot].mBufferState.acquireNotInQueue();
            } else {
                mSlots[slot].mBufferState.acquire();
            }
            mSlots[slot].mFence = Fence::NO_FENCE;
        }
 
 
        // If the buffer has previously been acquired by the consumer, set
        // mGraphicBuffer to NULL to avoid unnecessarily remapping this buffer
        // on the consumer side
        if (outBuffer-&gt;mAcquireCalled) {
            outBuffer-&gt;mGraphicBuffer = nullptr;
        }
 
 
        mCore-&gt;mQueue.erase(front);
 
 
        // We might have freed a slot while dropping old buffers, or the producer
        // may be blocked waiting for the number of buffers in the queue to
        // decrease.
        mCore-&gt;mDequeueCondition.notify_all();
 
 
        ATRACE_INT(mCore-&gt;mConsumerName.string(),
                static_cast&lt;int32_t&gt;(mCore-&gt;mQueue.size()));
#ifndef NO_BINDER
        mCore-&gt;mOccupancyTracker.registerOccupancyChange(mCore-&gt;mQueue.size());
#endif
        VALIDATE_CONSISTENCY();
    }
 
 
    if (listener != nullptr) {
        for (int i = 0; i &lt; numDroppedBuffers; &#43;&#43;i) {
            listener-&gt;onBufferReleased();
        }
    }
 
 
    return NO_ERROR;
}
```
